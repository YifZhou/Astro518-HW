RMS/\(\sqrt{\mu}\) changes with sample size

In [13]: 
rms = lambda lam, size: np.std(np.random.poisson(lam,size), ddof = 1)
sample_size = np.linspace(10, 20000, 1000)
rms_10 = map(lambda size: rms(10, size), sample_size)
rms_100 = map(lambda size: rms(100,size), sample_size)
fig, ax = plt.subplots()
ax.plot(sample_size, rms_10/np.sqrt(10), label = '$\mu = 10$')
ax.plot(sample_size, rms_100/np.sqrt(100), label = '$\mu = 100$')
ax.set_xlabel("Sample size")
ax.set_ylabel("$RMS/\sqrt{\mu}$")
ax.legend()
plt.show()
